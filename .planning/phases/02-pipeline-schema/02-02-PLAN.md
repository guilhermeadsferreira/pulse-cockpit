```typescript
function validateExternalSnapshot(data: unknown): RendererSnapshot | null {
  if (!data || typeof data !== 'object') return null
  const d = data as Record<string, unknown>
  // Must have atualizadoEm string
  if (typeof d.atualizadoEm !== 'string') return null
  return {
    atualizadoEm: d.atualizadoEm,
    jira: (d.jira && typeof d.jira === 'object') ? d.jira as RendererSnapshot['jira'] : null,
    github: (d.github && typeof d.github === 'object') ? d.github as RendererSnapshot['github'] : null,
    insights: Array.isArray(d.insights) ? d.insights as RendererSnapshot['insights'] : [],
  }
}
```
